# 媒體庫系統設計

## 概述
WebCube2027 的媒體庫系統基於 MinIO 物件儲存，提供安全、高效的多媒體檔案管理。

## MinIO 架構

### Bucket 設計
```typescript
// 固定少量 Bucket，簡化管理
const BUCKETS = {
```

## 安全存取機制

### 內部存取
```typescript
// 服務帳號 + Prefix 限制
class MinIO服務 {
  private client: MinioClient
  private domainPrefix: string
  
  constructor(domain: string) {
    this.client = new MinioClient({
      endPoint: 'minio.example.com',
      accessKey: 'service-account-key',
      secretKey: 'service-account-secret',
      useSSL: true
    })
    this.domainPrefix = `${domain}/`
  }
  
  // 只能存取自己 domain 的檔案
  async 取得檔案(key: string): Promise<Buffer> {
    if (!key.startsWith(this.domainPrefix)) {
      throw new Error('無權存取此檔案')
    }
    return this.client.getObject(BUCKETS.media, key)
  }
}
```

### 外部存取
```typescript
// Pre-signed URL（短時效）
class 媒體存取管理器 {
  async 生成存取URL(domain: string, key: string, 過期時間 = 3600): Promise<string> {
    const client = this.getMinIOClient(domain)
    return client.presignedGetObject(BUCKETS.media, key, 過期時間)
  }
  
  // 針對不同用途設定不同過期時間
  async 生成預覽URL(domain: string, key: string): Promise<string> {
    return this.生成存取URL(domain, key, 300) // 5 分鐘
  }
  
  async 生成下載URL(domain: string, key: string): Promise<string> {
    return this.生成存取URL(domain, key, 86400) // 24 小時
  }
}
```

## 媒體處理流程

### 上傳處理
```typescript
// 上傳 → 驗證 → 儲存 → 觸發處理
class 媒體上傳服務 {
  async 處理上傳(domain: string, 檔案: File): Promise<媒體資訊> {
    // 1. 驗證檔案
    await this.驗證檔案(檔案)
    
    // 2. 生成 UUID 和 Key
    const uuid = crypto.randomUUID()
    const key = this.生成檔案Key(domain, 檔案.type, uuid, 檔案.name)
    
    // 3. 上傳到 MinIO
    await this.上傳檔案(key, 檔案)
    
    // 4. 產生縮圖（非同步）
    await this.觸發縮圖生成(domain, uuid, 檔案.type)
    
    // 5. 記錄到資料庫
    const 媒體資訊 = await this.記錄媒體(domain, {
      uuid,
      原檔名: 檔案.name,
      類型: 檔案.type,
      大小: 檔案.size,
      原檔Key: key,
      縮圖Keys: []
    })
    
    return 媒體資訊
  }
}
```

### 縮圖生成
```typescript
// 非同步處理：上傳 → 佇列 → 轉換 → 儲存
class 縮圖處理器 {
  async 處理縮圖(domain: string, uuid: string, 檔案類型: string): Promise<void> {
    if (檔案類型 !== 'image') return
    
    // 1. 下載原檔
    const 原檔Buffer = await this.下載原檔(domain, uuid)
    
    // 2. 轉換尺寸
    const 縮圖Buffer = await this.轉換尺寸(原檔Buffer, 800) // 寬 800px
    
    // 3. 轉為 WebP
    const webpBuffer = await this.轉為WebP(縮圖Buffer)
    
    // 4. 上傳縮圖
    const 縮圖Key = this.生成縮圖Key(domain, uuid, 'thumb-800')
    await this.上傳縮圖(縮圖Key, webpBuffer)
    
    // 5. 更新資料庫
    await this.更新縮圖資訊(domain, uuid, 縮圖Key)
  }
  
  private async 轉換尺寸(buffer: Buffer, 目標寬度: number): Promise<Buffer> {
    // 使用 sharp 或其他影像處理庫
    return await sharp(buffer)
      .resize(目標寬度, null, { 
        withoutEnlargement: true,
        fit: 'inside'
      })
      .toBuffer()
  }
}
```

## 安全策略

### 檔案驗證
```typescript
class 檔案驗證器 {
  private 允許的MimeTypes = new Set([
    'image/jpeg', 'image/png', 'image/gif', 'image/webp',
    'video/mp4', 'video/webm',
    'audio/mp3', 'audio/wav', 'audio/ogg',
    'application/pdf', 'text/plain'
  ])
  
  async 驗證檔案(檔案: File): Promise<boolean> {
    // 1. 檢查副檔名
    if (!this.檢查副檔名(檔案.name)) {
      throw new Error('不支援的檔案格式')
    }
    
    // 2. 檢查 MIME 類型
    if (!this.允許的MimeTypes.has(檔案.type)) {
      throw new Error('不支援的檔案類型')
    }
    
    // 3. 檢查檔案大小
    if (檔案.size > this.取得大小限制(檔案.type)) {
      throw new Error('檔案超過大小限制')
    }
    
    // 4. 病毒掃描
    await this.病毒掃描(檔案)
    
    return true
  }
  
  private 取得大小限制(mimeType: string): number {
    const 限制 = {
      'image/': 10 * 1024 * 1024,      // 10MB
      'video/': 500 * 1024 * 1024,     // 500MB
      'audio/': 50 * 1024 * 1024,      // 50MB
      'application/': 100 * 1024 * 1024 // 100MB
    }
    
    for (const [prefix, size] of Object.entries(限制)) {
      if (mimeType.startsWith(prefix)) {
        return size
      }
    }
    
    return 10 * 1024 * 1024 // 預設 10MB
  }
}
```

### 存取控制
```typescript
class 存取控制器 {
  async 檢查存取權限(domain: string, 使用者ID: string, 檔案UUID: string): Promise<boolean> {
    // 1. 檢查檔案是否屬於該 domain
    const 檔案資訊 = await this.取得檔案資訊(檔案UUID)
    if (檔案資訊.domain !== domain) {
      return false
    }
    
    // 2. 檢查使用者權限
    const 使用者權限 = await this.取得使用者權限(使用者ID, domain)
    return 使用者權限.可讀取媒體
  }
  
  async 生成存取Token(domain: string, 使用者ID: string, 檔案UUID: string): Promise<string> {
    // JWT token 包含存取資訊
    return jwt.sign({
      domain,
      使用者ID,
      檔案UUID,
      用途: 'media_access',
      過期時間: Date.now() + 3600000 // 1 小時
    }, this.秘密金鑰)
  }
}
```

### CDN 整合
```typescript
class CDN管理器 {
  private cdnEndpoint = 'https://cdn.example.com'
  
  async 生成CDNURL(domain: string, key: string): Promise<string> {
    // 1. 檢查是否已存在 CDN
    if (await this.檢查CDN存在(key)) {
      return `${this.cdnEndpoint}/${key}`
    }
    
    // 2. 觸發 CDN 同步
    await this.觸發CDN同步(key)
    
    return `${this.cdnEndpoint}/${key}`
  }
  
  async 預熱熱門檔案(domain: string): Promise<void> {
    const 熱門檔案 = await this.取得熱門檔案(domain)
    for (const 檔案 of 熱門檔案) {
      await this.觸發CDN同步(檔案.key)
    }
  }
}
```

### 快取策略
```typescript
class 媒體快取管理器 {
  // 1. 記憶體快取：Metadata
  private metadataCache = new LRUCache<string, 媒體Metadata>({
    max: 1000,
    ttl: 1000 * 60 * 30 // 30 分鐘
  })
  
  // 2. Redis 快取：存取 URL
  async 快取存取URL(key: string, url: string, ttl = 3600): Promise<void> {
    await redis.setex(`media:url:${key}`, ttl, url)
  }
  
  // 3. 邊緣快取：HTTP Headers
  設定快取Headers(response: Response, 檔案類型: string): void {
  const 快取設定 = {
    'image': 'public, max-age=86400',      // 1 天
    'video': 'public, max-age=604800',     // 7 天
    'audio': 'public, max-age=604800',     // 7 天
    'file': 'public, max-age=3600'         // 1 小時
  }
  
  response.headers.set('Cache-Control', 快取設定[檔案類型])
  response.headers.set('ETag', this.生成ETag(檔案))
}
```
